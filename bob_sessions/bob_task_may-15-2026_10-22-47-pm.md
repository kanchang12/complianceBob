            job.current_stage = stage
            job.progress = progress
            job.message = message

        # Run analysis
        results = orchestrator.analyze_repository(
            github_url=job.github_url,

            job.error = str(e)

// Start analysis
const response = await fetch('http://localhost:5000/api/analyze', {

const {analysis_id} = await response.json();

const poll = setInterval(async () => {
    const status = await fetch(`http://localhost:5000/api/status/${analysis_id}`).then(r => r.json());
    console.log(`${status.progress}%: ${status.message}`);
    if (status.status === 'completed') {
        clearInterval(poll);
        const results = await fetch(`http://localhost:5000/api/results/${analysis_id}`).then(r => r.json());
        console.log('Results:', results);
