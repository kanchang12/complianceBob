429 | ### RAG System Implementation
430 | 
431 | The RAG system is crucial for boosting accuracy. Here's how it works:
432 | 
433 | ```python
434 | from sentence_transformers import SentenceTransformer
435 | import numpy as np
436 | import sqlite3
437 | 
438 | class RAGSystem:
439 |     """Retrieval-Augmented Generation system"""
440 |     
441 |     def __init__(self, db_path: str = "database/compliance.db"):
442 |         self.embedder = SentenceTransformer('all-MiniLM-L6-v2')
443 |         self.db_path = db_path
444 |         self.conn = sqlite3.connect(db_path)
445 |     
446 |     def process_documents(self, pdf_dir: str):
447 |         """Process compliance PDFs and create embeddings"""
448 |         from .pdf_processor import PDFProcessor
449 |         
450 |         processor = PDFProcessor()
451 |         
452 |         for pdf_file in Path(pdf_dir).glob("*.pdf"):
453 |             print(f"Processing {pdf_file.name}...")
454 |             
455 |             # Extract text from PDF
456 |             text = processor.extract_text(pdf_file)
457 |             
458 |             # Chunk into manageable pieces
459 |             chunks = self._chunk_text(text, chunk_size=1000)
460 |             
461 |             # Create embeddings
462 |             embeddings = self.embedder.encode(chunks)
463 |             
464 |             # Store in database
465 |             for chunk, embedding in zip(chunks, embeddings):
466 |                 self._store_embedding(
467 |                     document_id=pdf_file.stem,
468 |                     chunk_text=chunk,
469 |                     embedding=embedding
470 |                 )
471 |         
472 |         print(f"Processed {len(list(Path(pdf_dir).glob('*.pdf')))} documents")
473 |     
474 |     def search(self, query: str, top_k: int = 5) -> List[str]:
475 |         """
476 |         Search for relevant compliance context
477 |         
478 |         Args:
479 |             query: Search query (e.g., "GDPR data encryption")
480 |             top_k: Number of results to return
481 |         
482 |         Returns:
483 |             List of relevant text chunks
484 |         """
485 |         # Create query embedding
486 |         query_embedding = self.embedder.encode([query])[0]
487 |         
488 |         # Retrieve all embeddings from database
489 |         cursor = self.conn.cursor()
490 |         cursor.execute("""
491 |             SELECT chunk_text, embedding 
492 |             FROM embeddings
493 |         """)
494 |         
495 |         results = []
496 |         for row in cursor.fetchall():
497 |             chunk_text = row[0]
498 |             stored_embedding = np.frombuffer(row[1], dtype=np.float32)
499 |             
500 |             # Calculate cosine similarity
501 |             similarity = self._cosine_similarity(
502 |                 query_embedding, 
503 |                 stored_embedding
504 |             )
505 |             
506 |             results.append((similarity, chunk_text))
507 |         
508 |         # Sort by similarity and return top_k
509 |         results.sort(reverse=True, key=lambda x: x[0])
510 |         return [chunk for _, chunk in results[:top_k]]
511 |     
512 |     def _cosine_similarity(self, a: np.ndarray, 
513 |                           b: np.ndarray) -> float:
514 |         """Calculate cosine similarity between two vectors"""
515 |         return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))
516 |     
517 |     def _chunk_text(self, text: str, chunk_size: int = 1000, 
518 |                     overlap: int = 200) -> List[str]:
519 |         """Split text into overlapping chunks"""
520 |         chunks = []
521 |         start = 0
522 |         
523 |         while start < len(text):
524 |             end = start + chunk_size
525 |             chunk = text[start:end]
526 |             chunks.append(chunk)
527 |             start += chunk_size - overlap
528 |         
529 |         return chunks
530 | ```

781 | 
782 | ### Challenge 1: Context Window Limitations
783 | 
784 | **Problem**: Gemma 2B has an 8K token context window. Large files (10K+ lines) don't fit.
785 | 
786 | **Solution**: Implemented smart chunking strategy:
787 | ```python
788 | def chunk_file(content: str, max_tokens: int = 6000) -> List[str]:
789 |     """Split file into overlapping chunks"""
790 |     lines = content.split('\n')
791 |     chunks = []
792 |     current_chunk = []
793 |     current_tokens = 0
794 |     
795 |     for line in lines:
796 |         line_tokens = len(line.split())
797 |         
798 |         if current_tokens + line_tokens > max_tokens:
799 |             chunks.append('\n'.join(current_chunk))
800 |             current_chunk = current_chunk[-20:] + [line]
801 |             current_tokens = sum(len(l.split()) for l in current_chunk)
802 |         else:
803 |             current_chunk.append(line)
804 |             current_tokens += line_tokens
805 |     
806 |     return chunks
807 | ```
808 | 
809 | **Result**: Can now analyze files of any size with maintained context.
